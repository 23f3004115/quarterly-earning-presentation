# Simple NPV calculation example
def npv(rate, cashflows):
    total = 0.0
    for i, cf in enumerate(cashflows, start=1):
        total += cf / ((1 + rate) ** i)
    return total

cashflows = [100, 200, 300]
print("NPV @ 8%:", npv(0.08, cashflows))

\( \mathrm{WACC} = \dfrac{E}{D+E} \cdot r_e + \dfrac{D}{D+E} \cdot r_d \cdot (1 - t_c) \)
